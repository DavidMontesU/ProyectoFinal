
-- BASE DE DATOS: CLÍNICA VETERINARIA
-- Sistema de Control de Citas

Drop database veterinariadb;
CREATE DATABASE IF NOT EXISTS veterinariadb;
USE veterinariadb;


-- TABLA 1: TIPOS DE USUARIO
CREATE TABLE IF NOT EXISTS TiposUsuario(
    idTipo INT AUTO_INCREMENT PRIMARY KEY,
    descripcion VARCHAR(100) UNIQUE NOT NULL
);


-- TABLA 2: USUARIOS (Sistema de Login)

CREATE TABLE IF NOT EXISTS Usuarios(
    idUsuario INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    clave VARCHAR(50) NOT NULL,
    idTipo INT NOT NULL,
    FOREIGN KEY (idTipo) REFERENCES TiposUsuario(idTipo)
);

-- TABLA 3: ESPECIALIDADES VETERINARIAS

CREATE TABLE IF NOT EXISTS Especialidad(
    codEsp VARCHAR(5) PRIMARY KEY,
    descripcion VARCHAR(100) NOT NULL
);

-- TABLA 4: VETERINARIOS
CREATE TABLE IF NOT EXISTS Veterinario(
    idVet INT AUTO_INCREMENT PRIMARY KEY,
    codVet VARCHAR(5) UNIQUE NOT NULL,
    nombreVet VARCHAR(100) NOT NULL,
    codEsp VARCHAR(5) NOT NULL,
    telefono VARCHAR(20),
    edad INT,
    correo VARCHAR(100),
    FOREIGN KEY (codEsp) REFERENCES Especialidad(codEsp)
);

-- TABLA 5: CLIENTES/DUEÑOS
CREATE TABLE IF NOT EXISTS Cliente(
    idCliente INT AUTO_INCREMENT PRIMARY KEY,
    dni VARCHAR(8) UNIQUE NOT NULL,
    nombreCliente VARCHAR(100) NOT NULL,
    apellidoCliente VARCHAR(100) NOT NULL,
    telefono VARCHAR(20),
    direccion VARCHAR(200),
    correo VARCHAR(100)
);

-- TABLA 6: ESPECIES DE MASCOTAS
CREATE TABLE IF NOT EXISTS Especie(
    idEspecie INT AUTO_INCREMENT PRIMARY KEY,
    nombreEspecie VARCHAR(50) UNIQUE NOT NULL
);

-- TABLA 7: MASCOTAS
CREATE TABLE IF NOT EXISTS Mascota(
    idMascota INT AUTO_INCREMENT PRIMARY KEY,
    nombreMascota VARCHAR(100) NOT NULL,
    idEspecie INT NOT NULL,
    raza VARCHAR(100),
    edad INT,
    peso DECIMAL(5,2),
    idCliente INT NOT NULL,
    observaciones TEXT,
    FOREIGN KEY (idEspecie) REFERENCES Especie(idEspecie),
    FOREIGN KEY (idCliente) REFERENCES Cliente(idCliente)
);

-- TABLA 8: TIPOS DE SERVICIO
CREATE TABLE IF NOT EXISTS TipoServicio(
    idTipoServicio INT AUTO_INCREMENT PRIMARY KEY,
    descripcion VARCHAR(100) UNIQUE NOT NULL,
    precio DECIMAL(8,2) NOT NULL
);

-- TABLA 9: CITAS 
CREATE TABLE IF NOT EXISTS Cita(
    idCita INT AUTO_INCREMENT PRIMARY KEY,
    idMascota INT NOT NULL,
    idVet INT NOT NULL,
    idTipoServicio INT NOT NULL,
    fechaCita DATE NOT NULL,
    horaCita TIME NOT NULL,
    motivo TEXT,
    estado VARCHAR(20) DEFAULT 'Pendiente',
    observaciones TEXT,
    FOREIGN KEY (idMascota) REFERENCES Mascota(idMascota),
    FOREIGN KEY (idVet) REFERENCES Veterinario(idVet),
    FOREIGN KEY (idTipoServicio) REFERENCES TipoServicio(idTipoServicio)
);

-- TABLA 10: HISTORIAL CLÍNICO
CREATE TABLE IF NOT EXISTS HistorialClinico(
    idHistorial INT AUTO_INCREMENT PRIMARY KEY,
    idCita INT NOT NULL,
    diagnostico TEXT,
    tratamiento TEXT,
    receta TEXT,
    proximaCita DATE,
    fechaRegistro DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (idCita) REFERENCES Cita(idCita)
);

-- STORED PROCEDURES - VETERINARIOS
DELIMITER //
CREATE PROCEDURE sp_listarVeterinario()
BEGIN 
    SELECT v.idVet, v.codVet, v.nombreVet, v.codEsp,
           v.telefono, v.edad, v.correo,
           e.descripcion AS nomEspecialidad
    FROM Veterinario v
    INNER JOIN Especialidad e ON v.codEsp = e.codEsp;
END//
DELIMITER ;

DELIMITER //
CREATE PROCEDURE sp_registrarVeterinario(
    IN p_codVet VARCHAR(5),
    IN p_nomVet VARCHAR(100),
    IN p_codEsp VARCHAR(5),
    IN p_telf VARCHAR(20),
    IN p_edad INT,
    IN p_correo VARCHAR(100)
)
BEGIN
    INSERT INTO Veterinario(codVet, nombreVet, codEsp, telefono, edad, correo)
    VALUES (p_codVet, p_nomVet, p_codEsp, p_telf, p_edad, p_correo);
END//
DELIMITER ;

DELIMITER //
CREATE PROCEDURE sp_obtenerVeterinario(IN p_id INT)
BEGIN
    SELECT v.idVet, v.codVet, v.nombreVet, v.codEsp,
           v.telefono, v.edad, v.correo,
           e.descripcion AS nomEspecialidad
    FROM Veterinario v 
    INNER JOIN Especialidad e ON v.codEsp = e.codEsp
    WHERE v.idVet = p_id;
END//
DELIMITER ;

DELIMITER //
CREATE PROCEDURE sp_actualizarVeterinario(
    IN p_id INT,
    IN p_codVet VARCHAR(5),

    IN p_codEsp VARCHAR(5),
    IN p_telf VARCHAR(20),
    IN p_edad INT,
    IN p_correo VARCHAR(100)

BEGIN 
    UPDATE Veterinario 
    SET codVet = p_codVet, nombreVet = p_nomVet, codEsp = p_codEsp,
        telefono = p_telf, edad = p_edad, correo = p_correo

DELIMITER //
CREATE PROCEDURE sp_eliminarVeterinario(IN p_id INT)
BEGIN 
    DELETE FROM Veterinario WHERE idVet = p_id;
END//

DELIMITER //

-- STORED PROCEDURES - CITAS
DELIMITER //
CREATE PROCEDURE sp_listarCita()
BEGIN 
    SELECT c.idCita, c.idMascota, c.idVet, c.idTipoServicio,
           c.fechaCita, c.horaCita, c.motivo, c.estado, c.observaciones,
           m.nombreMascota,
           CONCAT(cl.nombreCliente, ' ', cl.apellidoCliente) AS nombreCliente,
           v.nombreVet,
           ts.descripcion AS tipoServicio,
           ts.precio
    FROM Cita c
    INNER JOIN Mascota m ON c.idMascota = m.idMascota
    INNER JOIN Cliente cl ON m.idCliente = cl.idCliente
    INNER JOIN Veterinario v ON c.idVet = v.idVet
    INNER JOIN TipoServicio ts ON c.idTipoServicio = ts.idTipoServicio
    ORDER BY c.fechaCita DESC, c.horaCita DESC;
END//
DELIMITER ;

END//
DELIMITER ;

CREATE PROCEDURE sp_obtenerHistorialByMascota(IN p_idMascota INT)
BEGIN
    SELECT h.idHistorial, h.idCita, h.diagnostico, h.tratamiento,
           h.receta, h.proximaCita, h.fechaRegistro,
           c.fechaCita, c.horaCita, c.motivo,

           ts.descripcion AS tipoServicio

    INNER JOIN Cita c ON h.idCita = c.idCita
    INNER JOIN Veterinario v ON c.idVet = v.idVet
    INNER JOIN TipoServicio ts ON c.idTipoServicio = ts.idTipoServicio
    WHERE c.idMascota = p_idMascota
    ORDER BY h.fechaRegistro DESC;
END//
DELIMITER ;
